from heapq import heappush, heappop

def dijkstra(start, end):
    heap = [(0, start)]
    distance = [999999999999] * (N + 1)  # 각 정점사이의 거리 무한대로 초기화
    distance[start] = 0  # 시작 지점 0으로 초기화

    while heap:
        pay, index = heappop(heap)
        for e, c in bus[index]:
            if distance[e] > pay + c:
                distance[e] = pay + c
                heappush(heap, (pay + c, e))
    return distance[end]

N = int(input())  # 도시
M = int(input())  # 버스
bus = [[] for _ in range(N + 1)]
for _ in range(M):
    start, end, cost = map(int, input().split())  # 출발지, 도착지, 비용
    bus[start].append((end, cost))
start, end = map(int, input().split())  # 찾고자하는 비용 경로(출발지, 도착지)


print(dijkstra(start, end))
